def bootstrap(data, func, reps, label):
	theta_hat_stars = []
	for i in range(reps):
		resample = choices(data, k=len(data))
		theta_hat_star = func(resample)
		theta_hat_stars.append(theta_hat_star)
	return pd.DataFrame(theta_hat_stars, columns=[label])
